5.5 Sample Automation Script

// tests/regression/auth/login.spec.js
import { test, expect } from '@playwright/test';
import { LoginPage } from '../../../pages/LoginPage.js';

test.describe('🔐 Login — Functional Tests', () => {

  test('REQ-007 | Valid login → dashboard redirect', async ({ page }) => {
    const login = new LoginPage(page);
    await login.navigate();
    await login.login('rahul@demo.com', 'Test@1234');
    await expect(page).toHaveURL(/dashboard\.html/);
  });

  test('REQ-008 | Account locks after 5 failed attempts', async ({ page }) => {
    const login = new LoginPage(page);
    await login.navigate();
    for (let i = 0; i < 5; i++) {
      await login.login('rahul@demo.com', 'WrongPass@1');
    }
    const alert = await login.getAlertText();
    expect(alert).toContain('locked');
  });

  test('REQ-015 | Back button after logout → stays on login', async ({ page }) => {
    await page.goto('/login.html');
    await login.login('rahul@demo.com', 'Test@1234');
    await page.click('button:has-text("Logout")');
    await page.goBack();
    await expect(page).toHaveURL(/login\.html/);
  });
});
